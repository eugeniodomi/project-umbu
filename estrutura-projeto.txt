projeto-umbu/
├── cypress/                # Testes end-to-end (E2E) da aplicação utilizando o Cypress
│   └── e2e/                # Cenários de teste automatizados simulando ações do usuário
├── public/                 # Arquivos estáticos servidos diretamente na raiz
├── src/                    # Código-fonte principal da aplicação React
│   ├── __tests__/          # Testes unitários e de integração utilizando o Vitest
│   ├── application/        # Camada de Aplicação: Orquestração de casos de uso (Use Cases) e fluxos
│   │   ├── services/       # Lógicas de coordenação entre domínio e infraestrutura
│   │   └── useCases/       # Regras específicas da aplicação que executam ações do usuário
│   ├── assets/             # Imagens, fontes e outros recursos estáticos
│   ├── components/         # Camada UI: Componentes visuais reutilizáveis (botões, modais, etc)
│   ├── context/            # Gerenciamento de estado global com React Context API
│   ├── domain/             # Camada de Domínio: Regras de negócio puras (independente de framework)
│   │   ├── models/         # Entidades de negócio, interfaces e tipos TypeScript
│   │   └── rules/          # Validações e lógicas centrais exclusivas do negócio
│   ├── hooks/              # Custom React Hooks para encapsular lógica de componentes UI
│   ├── infrastructure/     # Camada de Infraestrutura: Comunicação com o mundo externo
│   │   ├── api/            # Integrações com APIs externas, configuração de clientes HTTP
│   │   └── mock/           # Dados simulados localmente para desenvolvimento (JSONs)
│   ├── pages/              # Camada UI/Pages: Telas principais associadas a rotas da aplicação
│   │   ├── CartPage.tsx
│   │   ├── CheckoutPage.tsx
│   │   ├── FidelizacaoPage.tsx
│   │   ├── FranquiaSelectionPage.tsx
│   │   ├── LoginPage.tsx
│   │   └── MenuPage.tsx
│   ├── utils/              # Funções utilitárias auxiliares genéricas
│   ├── App.tsx             # Componente raiz que agrupa providers e sistema de rotas
│   └── main.tsx            # Ponto de entrada (entry point) principal do React no DOM
├── cypress.config.ts       # Arquivo de configuração base do Cypress (Testes E2E)
├── vitest.config.ts        # Arquivo de configuração base do Vitest (Testes Unitários)
├── vite.config.ts          # Configuração da ferramenta de build e servidor local Vite
├── package.json            # Definição das dependências do projeto e scripts de execução
└── README.md               # Documentação principal com instruções do projeto
